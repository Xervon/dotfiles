** Open Notmuch with (SPC m)
#+BEGIN_SRC emacs-lisp
(defun notmuch-search-unread ()
  (interactive)
  (notmuch-search "tag:unread" t))

(map! :leader
      :desc "Unread Mails" "m" #'notmuch-search-unread)
#+END_SRC
