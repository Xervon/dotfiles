** Open Notmuch with (SPC m)
#+BEGIN_SRC emacs-lisp
(defun my--notmuch-search-unread ()
  (interactive)
  (notmuch-search "tag:unread" t))

(map! :leader
      :desc "Unread Mails" "m" #'my--notmuch-search-unread)
#+END_SRC

      :desc "Find file in dotfiles" "e" #'my--find-file-in-dotfiles)
